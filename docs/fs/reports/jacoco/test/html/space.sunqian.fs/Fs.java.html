    /**
     * Casts and returns the given object as the specified type {@code T}. This method is equivalent to:
     * <pre>{@code

    @SuppressWarnings("noinspection unchecked")
    public static <T> T as(@Nullable Object obj) {
        return (T) obj;

    /**

     * @param <T> the type of the given object
     * @return the given object itself

    /**
     * Returns the default value if the given object is {@code null}, or the given object itself if it is not
     * {@code null}. This method is equivalent to:
     * <pre>{@code
     * return obj == null ? defaultValue : obj;
     * }</pre>
     * <p>
     * Note this method does not guarantee that the returned value must be {@code nonnull}.
     *
     * @param obj          the given object

        return obj == null ? defaultValue : obj;
    }

    /**
     * Returns the value computed from the specified supplier if the given object is {@code null}, or the given object

     * <pre>{@code
     * return obj == null ? supplier.get() : obj;
     * }</pre>
     * <p>
     * Note this method does not guarantee that the returned value must be {@code nonnull}.
     *
     * @param obj      the given object
     * @param supplier the specified supplier
     * @param <T>      the type of the returned value
     * @return the value computed from the specified supplier if the given object is {@code null}, or the given object
     * itself if it is not {@code null}

    /**
     * Runs the given action and wraps any {@link Exception} into a {@link RuntimeException}, the
     * {@link RuntimeException} is generated by the given {@link RuntimeException} generator. The logic as follows:
     * <pre>{@code
     * try {
     *     action.call();
     * } catch (Exception e) {
     *     throw unchecked.apply(e);
     * }
     * }</pre>
     *
     * @param action    the given action
     * @param unchecked the given {@link RuntimeException} generator
     */
    public static void uncheck(
        @Nonnull VoidCallable action,
        @Nonnull Function<? super @Nonnull Exception, ? extends @Nonnull RuntimeException> unchecked
    ) {
        try {
            action.call();
        } catch (Exception e) {
            throw unchecked.apply(e);
        }
    }

    /**
     * Runs the given action list sequentially and wraps any {@link Exception} into a {@link RuntimeException}, the
     * {@link RuntimeException} is generated by the given {@link RuntimeException} generator. This method collates all
     * exceptions into the first exception by {@link Exception#addSuppressed(Throwable)}. The behavior is as follows:
     * <pre>{@code
     * Exception firstException = null;
     * for (VoidCallable action : actions) {
     *     try {
     *         action.call();
     *     } catch (Exception e) {
     *         if (firstException == null) {
     *             firstException = e;
     *         } else {
     *             firstException.addSuppressed(e);
     *         }
     *     }
     * }
     * if (firstException != null) {
     *     throw unchecked.apply(firstException);
     * }
     * }</pre>
     *
     * @param actions   the given action list
     * @param unchecked the given {@link RuntimeException} generator
     */
    public static void uncheck(
        @Nonnull List<? extends @Nonnull VoidCallable> actions,
        @Nonnull Function<? super @Nonnull Exception, ? extends @Nonnull RuntimeException> unchecked
    ) {
        uncheck(actions, unchecked, false);
    }

    /**
     * Runs the given action list sequentially and wraps any {@link Exception} into a {@link RuntimeException}, the
     * {@link RuntimeException} is generated by the given {@link RuntimeException} generator. The
     * {@code throwImmediately} parameter controls whether to wrap the first exception immediately or collect all
     * exceptions into the first exception by {@link Exception#addSuppressed(Throwable)}. The logic as follows:
     * <pre>{@code
     * if (throwImmediately) {
     *     for (VoidCallable action : actions) {
     *         try {
     *             action.call();
     *         } catch (Exception e) {
     *             throw unchecked.apply(e);
     *         }
     *     }
     * } else {
     *     Exception firstException = null;
     *     for (VoidCallable action : actions) {
     *         try {
     *             action.call();
     *         } catch (Exception e) {
     *             if (firstException == null) {
     *                 firstException = e;
     *             } else {
     *                 firstException.addSuppressed(e);
     *             }
     *         }
     *     }
     *     if (firstException != null) {
     *         throw unchecked.apply(firstException);
     *     }
     * }
     * }</pre>
     *
     * @param actions          the given action list
     * @param unchecked        the given {@link RuntimeException} generator
     * @param throwImmediately whether to throw the first exception immediately
     */
    public static void uncheck(
        @Nonnull List<? extends @Nonnull VoidCallable> actions,
        @Nonnull Function<? super @Nonnull Exception, ? extends @Nonnull RuntimeException> unchecked,
        boolean throwImmediately
    ) {
        if (throwImmediately) {
            for (VoidCallable action : actions) {
                try {
                    action.call();
                } catch (Exception e) {
                    throw unchecked.apply(e);
                }
            }
        } else {
            Exception firstException = null;
            for (VoidCallable action : actions) {
                try {
                    action.call();
                } catch (Exception e) {
                    if (firstException == null) {
                        firstException = e;
                    } else {
                        firstException.addSuppressed(e);
                    }
                }
            }
            if (firstException != null) {
                throw unchecked.apply(firstException);
            }
        }
    }

    /**

    /**
     * Calls the given action and returns the result wrapped by {@link Ret}. The logic as follows:
     * <pre>{@code
     * try {
     *     T ret = action.call();
     *     return Ret.of(ret);
     * } catch (Throwable e) {
     *     return Ret.of(e);
     * }
     * }</pre>
     *
     * @param action the given action
     * @param <T>    the type of the result
     * @return the result of the given action wrapped by {@link Ret}

    public static <T> Ret<T> call(@Nonnull Callable<T> action) {
        try {
            T ret = action.call();
            return Ret.of(ret);
        } catch (Throwable e) {
            return Ret.of(e);
        }
    }

    /**
     * Executes the given task until it returns {@code true} or throws an exception. The original exception will be
     * wrapped by {@link AwaitingException} then thrown, using {@link AwaitingException#getCause()} can get the original
     * exception. The logic of this method is as follows:
     * <pre>{@code
     * try {
     *     while (true) {
     *         if (task.call()) {
     *             return;
     *         }
     *     }
     * } catch (Exception e) {
     *     throw new AwaitingException(e);
     * }
     * }</pre>
     * <p>
     * Note this method may cause high CPU usage. When the task determines to return {@code false}, consider adding some
     * measures (such as sleep the current thread in a very short time) to avoid it.
     *
     * @param task the given task to be executed
     * @throws AwaitingException if an error occurs while awaiting
     */
    public static void until(@Nonnull BooleanCallable task) throws AwaitingException {
        try {
            while (true) {
                if (task.call()) {
                    return;
                }
            }
        } catch (Exception e) {
            throw new AwaitingException(e);
        }
    }

    /**

     * <p>
     * The additional options parameter can be empty, in which case the default options will be used. Otherwise,

     * of which keys are same as the additional options will be overridden by those additional options.
     * <p>
     * This method is a shortcut to the {@link ObjectConverter#convert(Object, Class, Option[])}.
     *
     * @param src        the given source object
     * @param targetType the target type
     * @param options    the additional conversion options
     * @param <T>        the target type
     * @return the converted object, {@code null} is permitted
     * @throws UnsupportedObjectConvertException if the conversion to the target type is unsupported
     * @throws ObjectConvertException            if the conversion failed
     */

        @Nullable Object src,
